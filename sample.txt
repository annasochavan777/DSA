using System;
using System.IO;

class Program
{
    static void Main()
    {
        // Define the file path and the content to append
        string filePath = @"C:\example\output.txt";
        string contentToAppend = "This is the appended text.";

        // Append the content to the file
        try
        {
            // Append text to the file
            File.AppendAllText(filePath, contentToAppend + Environment.NewLine);

            Console.WriteLine("Text appended successfully.");
        }
        catch (Exception ex)
        {
            Console.WriteLine("An error occurred: " + ex.Message);
        }
    }
}
