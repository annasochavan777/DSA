using System;
using System.Text;
using System.Text.Json;

class Program
{
    static void Main()
    {
        // Example Base64 string
        string base64String = "eyJrZXkxIjoiVmFsdWUxIiwia2V5MiI6IlZhbHVlMiJ9"; // Base64 for {"key1":"Value1","key2":"Value2"}
        
        // Decode Base64 to JSON string
        string jsonString = Encoding.UTF8.GetString(Convert.FromBase64String(base64String));
        
        // Deserialize JSON to a Dictionary object
        var jsonObject = JsonSerializer.Deserialize<Dictionary<string, string>>(jsonString);

        // Add a new key-value pair
        jsonObject["key3"] = "Value3";

        // Serialize back to JSON string
        string updatedJsonString = JsonSerializer.Serialize(jsonObject);

        // Encode the updated JSON string back to Base64
        string updatedBase64String = Convert.ToBase64String(Encoding.UTF8.GetBytes(updatedJsonString));
        
        // Output the updated Base64 string
        Console.WriteLine($"Updated Base64 String: {updatedBase64String}");
    }
}
