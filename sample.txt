import difflib

# Define file paths
file1_path = r"C:\Path\To\File1.txt"
file2_path = r"C:\Path\To\File2.txt"
output_html_path = r"C:\Path\To\DifferenceReport.html"

# Read file contents
with open(file1_path, 'r') as file1:
    file1_lines = file1.readlines()

with open(file2_path, 'r') as file2:
    file2_lines = file2.readlines()

# Generate HTML diff
html_diff = difflib.HtmlDiff()
html_content = html_diff.make_file(file1_lines, file2_lines, fromdesc='File1', todesc='File2')

# Write HTML to file
with open(output_html_path, 'w') as output_file:
    output_file.write(html_content)

print(f"Beyond Compare-like report generated at: {output_html_path}")
