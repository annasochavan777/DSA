# Set your AutoSys command (example using autorep)
$autosysCommand = "autorep -J ALL -q"

# Run the command and get output
$jobsOutput = & cmd /c $autosysCommand

# Define output file path
$outputFilePath = "C:\Path\To\autosys_jobs.txt"

# Save the content as ANSI (Windows-1252)
[System.IO.File]::WriteAllLines($outputFilePath, $jobsOutput, [System.Text.Encoding]::GetEncoding("Windows-1252"))
