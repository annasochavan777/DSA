<#
.SYNOPSIS
    Processes a list of job names.

.DESCRIPTION
    This script accepts a list of job names as an input parameter and processes them.

.PARAMETER JobNames
    A list of job names to be processed.

.EXAMPLE
    .\JobsScript.ps1 -JobNames "Job1","Job2","Job3"

.NOTES
    Ensure job names are passed as an array.
#>

param (
    [string[]]$JobNames
)

# Process each job name
foreach ($job in $JobNames) {
    Write-Output "Processing job: $job"
}
