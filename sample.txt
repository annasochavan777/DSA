using System;
using System.Collections;
using System.Linq;

class Program
{
    static void Main()
    {
        // Create an ArrayList with mixed types
        ArrayList arrayList = new ArrayList();
        arrayList.Add("Apple");
        arrayList.Add(123.45m); // Decimal value
        arrayList.Add("Cherry");

        // Convert ArrayList to string[]
        string[] stringArray = arrayList.Cast<object>().Select(x => x.ToString()).ToArray();

        // Display the contents of string[]
        foreach (var item in stringArray)
        {
            Console.WriteLine(item);
        }
    }
}
