using System;
using System.IO;

class Program
{
    static void Main()
    {
        string filePath = @"C:\path\to\your\file.txt"; // Specify the file path
        string content = "This is the content to be written to the file."; // Content to write

        try
        {
            // Write the content to the file
            File.WriteAllText(filePath, content);
            Console.WriteLine("Content written successfully.");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"An error occurred: {ex.Message}");
        }
    }
}
