import difflib

# Define file paths
file1_path = r"C:\Path\To\web1.config"
file2_path = r"C:\Path\To\web2.config"
output_html_path = r"C:\Path\To\XMLComparisonReport.html"

# Read file contents
with open(file1_path, 'r', encoding='utf-8') as file1:
    file1_lines = file1.readlines()

with open(file2_path, 'r', encoding='utf-8') as file2:
    file2_lines = file2.readlines()

# Use HtmlDiff from difflib to generate the side-by-side HTML
html_diff = difflib.HtmlDiff(tabsize=4, wrapcolumn=80)
html_content = html_diff.make_file(file1_lines, file2_lines, fromdesc="web1.config", todesc="web2.config")

# Write the HTML content to the output file
with open(output_html_path, 'w', encoding='utf-8') as output_file:
    output_file.write(html_content)

print(f"Side-by-side comparison report generated at: {output_html_path}")
